    Abstraction Example
                            class Animal:
                            def __init__(self, species):
                                self.species = species
                            
                            def speak(self):
                                pass

                        class Dog(Animal):
                            def speak(self):
                                return "Woof!"

                        class Cat(Animal):
                            def speak(self):
                                return "Meow!"

                        dog = Dog('Canine')
                        cat = Cat('Feline')
                        print(dog.speak())  # Output: Woof!
                        print(cat.speak())  # Output: Meow!
    Encapsulation with Getters & Setters
                            class Enemy:
                            def __init__(self, type_of_enemy):
                                self.__type_of_enemy = type_of_enemy
                                
                            def get_type_of_enemy(self):
                                return self.__type_of_enemy
                                
                            def set_type_of_enemy(self, type_of_enemy):
                                self.__type_of_enemy = type_of_enemy

                        enemy = Enemy('Zombie')
                        print(enemy.get_type_of_enemy())  # Output: Zombie
    Inheritance Hierarchy
                            class Animal:
                            def __init__(self, weight, color, age):
                                self.weight = weight
                                self.color = color
                                self.age = age
                            
                            def eat(self):
                                print('Animal eating')

                        class Dog(Animal):
                            def __init__(self, weight, color, age, breed):
                                super().__init__(weight, color, age)
                                self.breed = breed
                            
                            def bark(self):
                                print('Woof!')

                        my_dog = Dog(20, 'brown', 3, 'Golden Retriever')
                        my_dog.eat()  # Inherited method
                        my_dog.bark()  # Dog-specific method
    Polymorphism in Action
                            class Animal:
                            def speak(self):
                                pass

                        class Dog(Animal):
                            def speak(self):
                                return "Woof!"

                        class Cat(Animal):
                            def speak(self):
                                return "Meow!"

                        def animal_sound(animal):
                            print(animal.speak())

                        # Same function handles different types
                        dog = Dog()
                        cat = Cat()
                        animal_sound(dog)  # Output: Woof!
                        animal_sound(cat)  # Output: Meow!
    Composition Example
                            class Engine:
                            def __init__(self, type):
                                self.type = type
                            
                            def start(self):
                                return f"{self.type} engine starting..."

                        class Car:
                            def __init__(self, model, engine_type):
                                self.model = model
                                self.engine = Engine(engine_type)
                            
                            def start_car(self):
                                return f"{self.model}: {self.engine.start()}"

                        tesla = Car("Model S", "Electric")
                        print(tesla.start_car())
